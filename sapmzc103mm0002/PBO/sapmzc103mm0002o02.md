*&---------------------------------------------------------------------*
*& Include          SAPMZC103MM0002O01

*&      Module  EXIT  INPUT
*&---------------------------------------------------------------------*

*----------------------------------------------------------------------*
MODULE exit INPUT.

  CALL METHOD : go_grid->free, go_cont->free.

  FREE : go_grid, go_cont.

  LEAVE TO SCREEN 0.

ENDMODULE.
*&---------------------------------------------------------------------*
*& Module GET_DATA OUTPUT
*&---------------------------------------------------------------------*
*&
*&---------------------------------------------------------------------*
MODULE get_data OUTPUT.
  IF gt_pohdback IS INITIAL.

    PERFORM get_main_data.
    PERFORM get_master_data.
    PERFORM set_master_data.

  ENDIF.
ENDMODULE.
*&---------------------------------------------------------------------*
*& Module INIT_DISPLAY OUTPUT
*&---------------------------------------------------------------------*
*&
*&---------------------------------------------------------------------*

  PERFORM init_display.
ENDMODULE.

*& Module STATUS_0110 OUTPUT
*&---------------------------------------------------------------------*
*&
*&---------------------------------------------------------------------*

  SET TITLEBAR 'TITLE110'.
ENDMODULE.

*& Module INIT_POP_DISPLAY OUTPUT

*&
*&---------------------------------------------------------------------*
MODULE init_pop_display OUTPUT.
